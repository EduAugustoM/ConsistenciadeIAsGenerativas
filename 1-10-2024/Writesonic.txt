Aqui está um código em Python que resolve o problema de encontrar números primos em um intervalo fornecido:

```python
def eh_primo(n):
    if n < 2:
        return False
    for i in range(2, int(n**0.5) + 1):
        if n % i == 0:
            return False
    return True

def encontrar_primos(inicio, fim):
    primos = []
    for num in range(max(2, inicio), fim + 1):
        if eh_primo(num):
            primos.append(num)
    return primos

# Exemplo de uso
inicio = 10
fim = 50
resultado = encontrar_primos(inicio, fim)
print(f"Números primos entre {inicio} e {fim}: {resultado}")
```
